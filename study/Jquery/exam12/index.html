2. 조건문

  • 특정 조건 만족시 해당 구문을 실행하고 싶을때 사용하는 구문.
  • 조건이 참이냐, 거짓이냐에 따라 문장을 실행하거나 실행하지 않을 수 있음.

1) if

  • if 단독문을 사용할 경우 참일때만 실행한다.
  • else if / else 로 분기처리하여 다중 조건문으로 사용 가능하다.
$(function(){
   var a = 3;
   if (a === 1) {
      alert('aaa');
   } else if (a === 2) {
      alert('bbb');
   } else {
      alert('ccc');
   }
});

2) switch

  • break문 : 반드시 넣어 주어야한다.(넣지 않는 경우 하단 조건문까지 실행)
  • default문 : 해당 사항이 없을때 기본값으로 실행되며, 생략가능
$(function(){
   var a = 21;
   switch (a) {
      case 10 :
         alert('10');
         break;
      case 21 :
         alert('21');
         break;
      default :
         alert('default');
         break;
   }
});

3. 반복문

특정 구문을 반복적으로 실행하고자 할 때 사용하는 구문

1) for

  • 특정 조건이 거짓으로 판별될 때까지 반복 실행된다.
  • 조건관련된 값은 초기값에 선언해준다. (성능이슈)
$(function(){
   for (var i = 0 ; i < 9 ; i++) {
      alert('aaa');
   }
});

2) while

  • 조건이 참일 경우 문장을 반복 실행하며, 해당 조건문이 거짓으로 판별된다면 실행을 멈춘 후 다음 문장으로 넘어간다.
$(function(){
   var a = 0;
   var b = 0;

   while (a < 3) {
      a++;
      b += a;
   }
});

(function (win, $){ //즉시 실행 구문
   'use strict';
   win.examProject = win.examProject || {};
   win.examProject.common = win.examProject.common || {};

   var UTIL = win.examProject.common.util;

   win.examProject.layerPopup = function () {
      this.init();
   };

   win.examProject.layerPopup.prototype = {
      init : function () {
         this.setElements();
         this.bindEvents();
      },
      setElements : function () {
         this.btn = $('.btn');
      },
      bindEvents : function () {
         this.btn.on('click', $.proxy(this.viewFunc, this));
      },
      viewFunc : function (e) {
         $(e.currentTarget).parent().addClass('active');
      }
   }
   $(function () {
        win.examLayerPopup = new win.examProject.layerPopup();
    });
})(window, window.jQuery)
