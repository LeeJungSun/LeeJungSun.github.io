2) Code

<div class="exam__box">
    <a href="https://www.naver.com/" target="_blank" class="link">a 태그 입니당!!</a>
    <div class="btn_area">
        <button type="button" id="linkButton"><span>trigger button</span></button>
    </div>
</div>
(function($, win, doc) {
    'use strict';
    var example = {
        opts : {
            obj : '.exam__box',
            link : '.link',
            btnArea : '.btn_area',
            btn : '#linkButton'
        },
        init : function () {
            this.setElements();
            this.bindEvents();
        },
        setElements : function () {
            this.obj = $(this.opts.obj);
            this.btnArea = this.obj.find(this.opts.btnArea);
            this.btn = this.btnArea.find(this.opts.btn);
            this.link = this.obj.find(this.opts.link);
        },
        bindEvents : function () {
            this.btn.on('click', $.proxy(this.clickFunc, this));
            this.link.on('click', $.proxy(this.linkClickFunc, this));
        },
        clickFunc : function (e) {
            // link trigger
            this.link.trigger('click');
        },
        linkClickFunc : function () {
            alert('link click!');
        }
    }
    example.init();
})(jQuery, window, document);
