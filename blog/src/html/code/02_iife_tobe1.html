(function () {
    var obj = document.getElementById('test');
        btn = obj.getElementsByTagName('button');
    function addFunc (index) {
        return function () {
            alert(index)
        }
    }
    for (var i = 0, max = btn.length; i < max; i++) {
        btn[i].onclick = addFunc(i);
    }
})();
